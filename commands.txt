sudo certbot --nginx -d pacs.radport.net -d backend.radport.net

server {
    listen 80;
    server_name pacs.radport.net;
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl http2;
    server_name pacs.radport.net;

    ssl_certificate     /etc/letsencrypt/live/pacs.radport.net/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/pacs.radport.net/privkey.pem;
    ssl_protocols       TLSv1.2 TLSv1.3;
    ssl_ciphers         HIGH:!aNULL:!MD5;
    ssl_prefer_server_ciphers on;

    add_header Strict-Transport-Security "max-age=63072000; includeSubDomains" always;
    add_header X-Content-Type-Options nosniff   always;
    add_header X-Frame-Options          DENY    always;
    add_header Referrer-Policy strict-origin-when-cross-origin always;

    # DICOM studies and STOW-RS uploads can be very large
    client_max_body_size  4G;
    client_body_buffer_size 1M;

    proxy_read_timeout    600s;
    proxy_send_timeout    600s;
    proxy_connect_timeout  60s;

    # Optional: restrict the built-in pacs Explorer UI to known IPs
    # location ~ ^/(app|ui)/ {
    #     allow 1.2.3.4;
    #     deny all;
    #     proxy_pass http://127.0.0.1:8042;
    # }

    location / {
        proxy_pass http://127.0.0.1:8042;
        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_request_buffering off;   # stream large STOW uploads
        proxy_http_version 1.1;
    }
}


/etc/nginx/sites-available/backend.radport

server {
    listen 80;
    server_name backend.radport.net;
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl http2;
    server_name backend.radport.net;

    ssl_certificate     /etc/letsencrypt/live/backend.radport.net/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/backend.radport.net/privkey.pem;
    ssl_protocols       TLSv1.2 TLSv1.3;
    add_header Strict-Transport-Security "max-age=63072000; includeSubDomains" always;

    location /static/ { alias /var/www/radport/static/; }
    location /media/  { alias /var/www/radport/media/;  }

    location / {
        proxy_pass http://unix:/run/gunicorn/radport.sock;
        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}




sudo ln -s /etc/nginx/sites-available/pacs /etc/nginx/sites-enabled/
sudo ln -s /etc/nginx/sites-available/radport /etc/nginx/sites-enabled/
sudo nginx -t && sudo systemctl reload nginx
